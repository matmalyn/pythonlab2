n = int(input('Podaj  ilosc wprowadzanych liczbe : '))
nList = []

i = 1
while i <= n:
    c = input('Podaj liczbê:')
    nList.append(c)
    i += 1

print("Oto lista liczb:")
print(nList)


print("Zawartoœæ listy ([indeks] wartoœæ):")
for i, v in enumerate(nList):
    print("[", i, "]", v)

print("Elementy w odwróconym porz¹dku:")
for e in reversed(nList):
    print(e, end=" ")

print()
print("Elementy posortowane rosn¹co:")
for e in sorted(nList):
    print(e, end=" ")

print("Usuniecie elementu ")
toDelete = input("Który element usunac?:")
nList.remove(toDelete)
print(nList)

print("Wyszukiwanie i zliczanie elementu w liœcie")
e = int(input("Podaj liczbê: "))
print("Liczba wyst¹pieñ: ")
print(nList.count(e))
print("Indeks pierwszego wyst¹pienia: ")
if nList.count(e):
    print(nList.index(e))
else:
    print("Brak elementu w liœcie")
